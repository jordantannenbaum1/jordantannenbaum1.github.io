What it is
HOLC redlining grades from the 1930s joined to 2024 rent burden and homeownership data at the census tract level. Does where a neighborhood was graded 90 years ago still predict who can afford to live there today?

-- int_tract_holc_join.sql
with crosswalk as (
    select * from {{ ref('stg_holc_tract_crosswalk') }}
),
acs as (
    select * from {{ ref('stg_acs') }}
)
select
    a.tract_geoid, a.city, a.state,
    c.holc_grade,
    c.overlap_pct,
    a.rent_burden_pct,
    a.median_hh_income
from acs a
left join crosswalk c using (tract_geoid)
where c.overlap_pct >= 0.5
