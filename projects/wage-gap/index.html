-- int_gap_calcs.sql
with by_group as (
    select
        industry_group, race_category, gender, education_level,
        median(weekly_earnings)  as median_earnings,
        count(*)               as n_workers
    from {{ ref('int_worker_enriched') }}
    where full_time_flag = true and weekly_earnings > 0
    group by 1,2,3,4
),
baseline as (
    select industry_group, education_level,
           median_earnings as baseline_earnings
    from by_group
    where race_category = 'White' and gender = 'Male'
)
select
    g.*,
    b.baseline_earnings,
    round(g.median_earnings / b.baseline_earnings, 4) as earnings_ratio
from by_group g
left join baseline b using (industry_group, education_level)
